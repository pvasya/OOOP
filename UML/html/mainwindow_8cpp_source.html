89void MainWindow::on_savebtn_clicked()
90{
91 if (ui->lineEdit->text().isEmpty()) {
92 QMessageBox msgBoxEmpty;
93 msgBoxEmpty.setObjectName("msgBox");
94 msgBoxEmpty.setIcon(QMessageBox::Warning);
95 msgBoxEmpty.setText("Please name the note");
96 msgBoxEmpty.setGeometry(this->x() + 50, this->y() + 50,300,480);
97 msgBoxEmpty.setStyleSheet("background-color: rgb(253, 223, 70);");
98 msgBoxEmpty.setWindowFlags(Qt::FramelessWindowHint);
99 msgBoxEmpty.exec();
100 return;
101 }
102
103 QString filePath = QDir::homePath() + "/Desktop/mynotesmaker.json";
104 nlohmann::json data;
105
106 if (QFile::exists(filePath)) {
107 QFile existingFile(filePath);
108 if (existingFile.open(QFile::ReadOnly | QFile::Text)) {
109 QTextStream in(&existingFile);
110 QString jsonString = in.readAll();
111 existingFile.close();
112 try {
113 data = nlohmann::json::parse(jsonString.toStdString());
114 } catch (const std::exception& e) {
115 qWarning() << "Error parsing JSON: " << e.what();
116 return;
117 }
118 std::string noteName = ui->lineEdit->text().toStdString();
119 for (const auto& note : data) {
120 if (note["name"] == noteName) {
121 QMessageBox msgBoxSame;
122 msgBoxSame.setObjectName("msgBox");
123 msgBoxSame.setIcon(QMessageBox::Warning);
124 msgBoxSame.setText("A note with the same name already exists.\nPlease choose a different name.");
125 msgBoxSame.setGeometry(this->x(), this->y() + 50, 300, 480);
126 msgBoxSame.setStyleSheet("background-color: rgb(253, 223, 70);");
127 msgBoxSame.setWindowFlags(Qt::FramelessWindowHint);
128 msgBoxSame.exec();
129 return;
130 }
131 }
132 } else {
133 qWarning() << "Error reading the existing file.";
134 return;
135 }
136 }
137
138 nlohmann::json note;
139 note["name"] = ui->lineEdit->text().toStdString();
140 note["text"] = ui->plainTextEdit->toPlainText().toStdString();
141 note["font"] = ui->plainTextEdit->font().toString().toStdString();
142
143 data.push_back(note);
144
145 QFile file(filePath);
146 if (file.open(QFile::WriteOnly | QFile::Text)) {
147 QTextStream out(&file);
148 out << QString::fromStdString(data.dump());
149 file.close();
150 } else {
151 qWarning() << "ERROR";
152 }
153}

171 nlohmann::json jsonRequest;
